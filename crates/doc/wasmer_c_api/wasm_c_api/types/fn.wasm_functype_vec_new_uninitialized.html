Expand description

Creates a new uninitialized vector of [wasm_functype_t *].

Example

int main() {
    // Creates an empty vector of `wasm_functype_t *`.
    wasm_functype_vec_t vector;
    wasm_functype_vec_new_uninitialized(&vector, 3);

    // Check that it contains 3 items.
    assert(vector.size == 3);

    // Free it.
    wasm_functype_vec_delete(&vector);

    return 0;
}
